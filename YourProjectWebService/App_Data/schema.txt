CREATE TABLE IF NOT EXISTS Brand (
	BrandId INTEGER PRIMARY KEY AUTOINCREMENT,
	brandName TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS Tool (
	ToolId INTEGER PRIMARY KEY AUTOINCREMENT,
	brandId INTEGER NOT NULL,
	description TEXT NOT NULL,
	active NUMERIC NOT NULL,
	comments TEXT NOT NULL,
	inUse NUMERIC NOT NULL,
	FOREIGN KEY (brandId) REFERENCES Brand(BrandId)
);

CREATE TABLE IF NOT EXISTS PatronToolLoan (
	PatronToolLoanId INTEGER PRIMARY KEY AUTOINCREMENT,
	toolId INTEGER NOT NULL,
	patronId INTEGER NOT NULL,
	dateRented NUMERIC NOT NULL,
	dateReturned NUMERIC,
	workspace TEXT NOT NULL,
	FOREIGN KEY (toolId) REFERENCES Tool(ToolId),
	FOREIGN KEY (patronId) REFERENCES Patron(PatronId)
);

CREATE TABLE IF NOT EXISTS Patron (
	PatronId INTEGER PRIMARY KEY AUTOINCREMENT,
	patronName TEXT NOT NULL,
	isGroup NUMERIC NOT NULL
);

CREATE TABLE IF NOT EXISTS Staff (
	StaffId INTEGER PRIMARY KEY AUTOINCREMENT,
	userName TEXT NOT NULL,
	password TEXT NOT NULL
);
